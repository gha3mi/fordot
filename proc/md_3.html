   pure function md_3(u,v) result(a)
      real(rk), dimension(:), intent(in), contiguous :: u, v
      real(rk) :: a
      integer :: i

      a = 0.0_rk
      do i = 1, size(u)
         a = a + u(i)*v(i)
      end do
   end function md_3
